cmake_minimum_required(VERSION 3.16)

project(KURS VERSION 0.1 LANGUAGES CXX)

set(CMAKE_AUTOUIC ON)
set(CMAKE_AUTOMOC ON)
set(CMAKE_AUTORCC ON)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# –®—É–∫–∞—î–º–æ Qt6. –Ø–∫—â–æ –∑–Ω–∞–π–¥–µ–Ω–æ, CMake –∞–≤—Ç–æ–º–∞—Ç–∏—á–Ω–æ –≤–∏–∑–Ω–∞—á–∏—Ç—å QT_VERSION_MAJOR, QT_VERSION —Ç–æ—â–æ.
# üí° –í–ò–ü–†–ê–í–õ–ï–ù–û: –ü–æ–≤–µ—Ä—Ç–∞—î–º–æ LinguistTools –¥–æ find_package, —â–æ–± –∫–æ–º–∞–Ω–¥–∞ qt_create_translation –±—É–ª–∞ –≤—ñ–¥–æ–º–æ—é.
find_package(Qt6 REQUIRED COMPONENTS Core Gui Qml Quick Sql LinguistTools)

set(TS_FILES KURS_uk_UA.ts)

set(PROJECT_SOURCES
    main.cpp
    Database.cpp
    Database.h
    ${TS_FILES}
)

# üí° –í–ò–ü–†–ê–í–õ–ï–ù–ù–Ø: –ó–ê–õ–ò–®–ê–Ñ–ú–û –õ–ò–®–ï –ö–û–ù–§–Ü–ì–£–†–ê–¶–Ü–Æ –î–õ–Ø QT 6

qt_add_executable(KURS
    MANUAL_FINALIZATION
    ${PROJECT_SOURCES}
    MForm.qml
    build/Desktop_Qt_6_10_0_MinGW_64_bit-Debug/MForm.qml
    Database.cpp main.cpp
    main.cpp
    Database.cpp main.cpp
    Database.cpp main.cpp
)

# qt_create_translation —Ç–µ–ø–µ—Ä –ø–æ–≤–∏–Ω–µ–Ω –ø—Ä–∞—Ü—é–≤–∞—Ç–∏, –æ—Å–∫—ñ–ª—å–∫–∏ LinguistTools –∑–Ω–∞–π–¥–µ–Ω–æ.
qt_create_translation(QM_FILES ${CMAKE_SOURCE_DIR} ${TS_FILES})

# –í–∏–∫–æ—Ä–∏—Å—Ç–æ–≤—É—î–º–æ —Ü—ñ–ª—ñ Qt6::...
set(QT_TARGETS
    Qt6::Core
    Qt6::Gui
    Qt6::Qml
    Qt6::Quick
    Qt6::Sql
)


# –õ–Ü–ù–ö–£–í–ê–ù–ù–Ø: –ü—ñ–¥–∫–ª—é—á–∞—î–º–æ –≤—Å—ñ –Ω–µ–æ–±—Ö—ñ–¥–Ω—ñ –±—ñ–±–ª—ñ–æ—Ç–µ–∫–∏
# –£–í–ê–ì–ê: LinguistTools —Ç—É—Ç –≤—ñ–¥—Å—É—Ç–Ω—ñ–π, –æ—Å–∫—ñ–ª—å–∫–∏ –≤—ñ–Ω –≤–∏–∫–ª–∏–∫–∞–≤ –ø–æ–º–∏–ª–∫—É –ª—ñ–Ω–∫—É–≤–∞–Ω–Ω—è.
target_link_libraries(KURS PRIVATE ${QT_TARGETS})


# –ù–∞–ª–∞—à—Ç—É–≤–∞–Ω–Ω—è –≤–ª–∞—Å—Ç–∏–≤–æ—Å—Ç–µ–π —Ü—ñ–ª—ñ
# üí° –í–ò–ü–†–ê–í–õ–ï–ù–û: –í–∏–¥–∞–ª—è—î–º–æ –Ω–µ–æ–±–æ–≤'—è–∑–∫–æ–≤–∏–π –±–ª–æ–∫ –¥–ª—è Mac OSX.

set_target_properties(KURS PROPERTIES
    MACOSX_BUNDLE_BUNDLE_VERSION ${PROJECT_VERSION}
    MACOSX_BUNDLE_SHORT_VERSION_STRING ${PROJECT_VERSION_MAJOR}.${PROJECT_VERSION_MINOR}
    MACOSX_BUNDLE TRUE
    WIN32_EXECUTABLE TRUE
)

include(GNUInstallDirs)
install(TARGETS KURS
    BUNDLE DESTINATION .
    LIBRARY DESTINATION ${CMAKE_INSTALL_LIBDIR}
    RUNTIME DESTINATION ${CMAKE_INSTALL_BINDIR}
)

# –§—ñ–Ω–∞–ª—ñ–∑—É—î–º–æ –∑–±—ñ—Ä–∫—É –¥–ª—è Qt 6
qt_finalize_executable(KURS)
